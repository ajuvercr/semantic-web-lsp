32async fn extract_file_url(prefix: &str, client: &impl Client) -> Option<String> {
33    let url = format!(
34        "https://lov.linkeddata.es/dataset/lov/api/v2/vocabulary/info?vocab={}",
35        prefix
36    );
37    match client.fetch(&url, &std::collections::HashMap::new()).await {
38        Ok(resp) if resp.status == 200 => match serde_json::from_str::<Vocab>(&resp.body) {
39            Ok(x) => {
40                let versions: Vec<_> = x.versions.iter().flat_map(|x| &x.file_url).collect();
41                debug!(
42                    "Found lov response ({} versions) {:?}",
43                    x.versions.len(),
44                    versions
45                );
46                x.versions
47                    .into_iter()
48                    .flat_map(|x| x.file_url.map(|url| (url, x.issued)))
49                    .max_by_key(|x| x.1)
50                    .map(|x| x.0)
51            }
52            Err(e) => {
53                error!("Deserialize failed ({}) {:?}", url, e);
54                None
55            }
56        },
57        Ok(resp) => {
58            error!("Fetch ({}) failed status {}", url, resp.status);
59            None
60        }
61        Err(e) => {
62            error!("Fetch ({}) failed {:?}", url, e);
63            None
64        }
65    }
66}

81pub fn open_imports(
82    query: Query<(&Triples, &RopeC), Changed<Triples>>,
83    mut opened: Local<HashSet<String>>,
84    sender: Res<CommandSender>,
85) {
86    for (triples, _) in &query {
87        for object in triples
88            .quads_matching(Any, [owl::imports], Any, Any)
89            .flatten()
90            .flat_map(|s| s.o().iri())
91            .flat_map(|s| Url::parse(s.as_str()))
92        {
93            if opened.contains(object.as_str()) {
94                continue;
95            }
96            opened.insert(object.as_str().to_string());
97
98            #[cfg(not(target_arch = "wasm32"))]
99            if let Some(content) = object
100                .to_file_path()
101                .ok()
102                .and_then(|p| read_to_string(p).ok())
103            {
104                spawn_document(object, content, &sender.0, None);
105            }
106        }
107    }
108}

110/// First of al, fetch the lov dataset information at url https://lov.linkeddata.es/dataset/lov/api/v2/vocabulary/info?vocab=${prefix}
111/// Next, extract that json object into an object and find the latest dataset
112pub fn fetch_lov_properties<C: Client + Resource>(
113    sender: Res<CommandSender>,
114    query: Query<
115        &Prefixes,
116        (
117            Or<((Changed<Prefixes>, With<Open>), Changed<Open>)>,
118            // Without<Dirty>,
119        ),
120    >,
121    mut prefixes: Local<HashSet<String>>,
122    client: Res<C>,
123    mut helper: ResMut<LovHelper>,
124    cache: Res<Cache>,
125) {
126    // println!("fetch lov properties");
127    for prefs in &query {
128        // println!("Found some turtle!");
129        for prefix in prefs.0.iter() {
130            if let Some(e) = helper.has_entry_mut(&prefix) {
131                let name = e.name();
132                if !prefixes.contains(&name) {
133                    prefixes.insert(name);
134                    from_cache(&e, &cache, &sender);
135                }
136            } else {
137                let e = helper.create_entry(&prefix);
138                let name = e.name();
139                if !prefixes.contains(&name) {
140                    prefixes.insert(name);
141                    if let Some(url) = e.url(&cache) {
142                        // let prefix = prefix.prefix.0.clone();
143                        if let Some(local) = lov::LOCAL_PREFIXES
144                            .iter()
145                            .find(|x| x.location == prefix.url.as_str())
146                        {
147                            debug!("Local lov");
148                            local_lov(local, url, &sender);
149                        } else {
150                            debug!("Remove lov");
151                            let sender = sender.0.clone();
152                            let c = client.as_ref().clone();
153                            client.spawn(fetch_lov(prefix.clone(), url, c, sender));
154                        }
155                    } else {
156                        debug!("Failed to find url");
157                    }
158                } else {
159                    debug!("Prefixes is already present {}", name);
160                }
161            }
162        }
163    }
164}

176type Sender = futures::channel::mpsc::UnboundedSender<CommandQueue>;
177fn spawn_document(url: Url, content: String, sender: &Sender, from_prefix: Option<FromPrefix>) {
178    let mut command_queue = CommandQueue::default();
179    let item = TextDocumentItem {
180        version: 1,
181        uri: url.clone(),
182        language_id: String::from("turtle"),
183        text: String::new(),
184    };
185
186    let spawn = spawn_or_insert(
187        url.clone(),
188        (
189            RopeC(ropey::Rope::from_str(&content)),
190            Source(content),
191            Label(url), // this might crash
192            Wrapped(item),
193            Types(HashMap::new()),
194        ),
195        Some("turtle".into()),
196        (),
197    );
198
199    command_queue.push(move |world: &mut World| {
200        let span = span!(tracing::Level::INFO, "span lov");
201        let _enter = span.enter();
202        let e = spawn(world);
203        if let Some(from) = from_prefix {
204            world.entity_mut(e).insert(from);
205        }
206        world.run_schedule(ParseLabel);
207        drop(_enter);
208    });
209
210    let _ = sender.unbounded_send(command_queue);
211}

213async fn fetch_lov<C: Client + Resource>(prefix: Prefix, label: Url, c: C, sender: Sender) {
214    if let Some(url) = extract_file_url(&prefix.prefix, &c).await {
215        match c.fetch(&url, &std::collections::HashMap::new()).await {
216            Ok(resp) if resp.status == 200 => {
217                spawn_document(label, resp.body, &sender, Some(FromPrefix(prefix)));
218            }
219            Ok(resp) => {
220                error!("Fetch ({}) failed status {}", url, resp.status);
221            }
222            Err(e) => {
223                error!("Fetch ({}) failed {:?}", url, e);
224            }
225        }
226    }
227}

304#[derive(Debug, Resource)]
305pub struct OntologyExtractor {
306    quads: Vec<MyQuad<'static>>,
307    properties: Vec<MyTerm<'static>>,
308    classes: Vec<MyTerm<'static>>,
309}
310
311struct LocalMatcher<'a> {
312    properties: &'a [MyTerm<'static>],
313}
314
315impl TermMatcher for LocalMatcher<'_> {
316    type Term = MyTerm<'static>;
317
318    fn matches<T2: sophia_api::prelude::Term + ?Sized>(&self, term: &T2) -> bool {
319        for p in self.properties {
320            if term.eq(p) {
321                return false;
322            }
323        }
324
325        true
326    }
327}
328
329impl OntologyExtractor {
330    pub fn new() -> Self {
331        Self {
332            quads: vec![],
333            classes: vec![MyTerm::<'static>::named_node(
334                "http://www.w3.org/2000/01/rdf-schema#Class",
335                0..1,
336            )],
337            properties: vec![MyTerm::<'static>::named_node(
338                "http://www.w3.org/1999/02/22-rdf-syntax-ns#Property",
339                0..1,
340            )],
341        }
342    }
343
344    pub fn properties<'a>(&'a self) -> &'a [MyTerm<'static>] {
345        &self.properties[..]
346    }
347
348    pub fn classes<'a>(&'a self) -> &'a [MyTerm<'static>] {
349        &self.classes[..]
350    }
351
352    fn extract_step(quads: &Vec<MyQuad<'static>>, items: &mut Vec<MyTerm<'static>>) -> bool {
353        let new_items: Vec<_> = quads
354            .quads_matching(
355                LocalMatcher { properties: &items },
356                [rdfs::subClassOf],
357                &items[..],
358                Any,
359            )
360            .flatten()
361            .map(|x| x.to_s().to_owned())
362            .collect();
363
364        let added = !new_items.is_empty();
365        items.extend(new_items);
366        added
367    }
368
369    fn extract(&mut self) {
370        loop {
371            if !OntologyExtractor::extract_step(&self.quads, &mut self.properties) {
372                break;
373            }
374        }
375
376        loop {
377            if !OntologyExtractor::extract_step(&self.quads, &mut self.classes) {
378                break;
379            }
380        }
381    }
382}

390#[derive(Debug, Clone, Serialize, Deserialize)]
391pub struct LovEntry {
392    prefix: String,
393    url: String,
394    num: usize,
395    state: EntryState,
396}

398impl LovEntry {
399    fn name(&self) -> String {
400        format!("{}-{}.ttl", self.num, self.prefix)
401    }
402    pub fn url(&self, cache: &Cache) -> Option<Url> {
403        self.file_url(cache).or_else(|| self.remote_url())
404    }
405
406    #[cfg(not(target_arch = "wasm32"))]
407    fn file_url(&self, cache: &Cache) -> Option<Url> {
408        let p = cache.path()?;
409        let url = p.join(self.name());
410        Url::from_file_path(url).ok()
411    }
412
413    #[cfg(target_arch = "wasm32")]
414    fn file_url(&self, cache: &Cache) -> Option<Url> {
415        None
416    }
417
418    fn remote_url(&self) -> Option<Url> {
419        Url::from_str(&self.url).ok()
420    }
421
422    pub fn save(&mut self, cache: &Cache, content: &str) -> Option<()> {
423        self.state = EntryState::Ready;
424        cache.write_file(&self.name(), content)
425    }
426}

428#[derive(Debug, Clone, Serialize, Deserialize, Resource)]
429pub struct LovHelper {
430    entries: Vec<LovEntry>,
431}
432
433impl LovHelper {
434    fn try_from_cache(cache: &Cache) -> Option<Self> {
435        let c = cache.get_file("index.json")?;
436        info!("Found index file! {}", c);
437        serde_json::from_str(&c).ok()
438    }
439
440    pub fn from_cache(cache: &Cache) -> Self {
441        Self::try_from_cache(cache).unwrap_or_else(|| Self {
442            entries: Vec::new(),
443        })
444    }
445
446    pub fn save(mut self, cache: &Cache) -> Option<()> {
447        self.entries = self
448            .entries
449            .into_iter()
450            .filter(|x| x.state == EntryState::Ready)
451            .collect();
452        let st = serde_json::to_string(&self).ok()?;
453        info!("Save index file! {}", st);
454        cache.write_file("index.json", &st)
455    }
456
457    pub fn has_entry_mut(&mut self, prefix: &Prefix) -> Option<&mut LovEntry> {
458        self.entries
459            .iter_mut()
460            .find(|e| e.prefix == prefix.prefix && e.url == prefix.url.as_str())
461    }
462
463    pub fn has_entry(&self, prefix: &Prefix) -> Option<&LovEntry> {
464        self.entries
465            .iter()
466            .find(|e| e.prefix == prefix.prefix && e.url == prefix.url.as_str())
467    }
468
469    pub fn create_entry(&mut self, prefix: &Prefix) -> &LovEntry {
470        debug!("Create entry for {:?}", prefix);
471        if let Some(e) = self.entries.iter().enumerate().find_map(|(i, e)| {
472            (e.prefix == prefix.prefix && e.url == prefix.url.as_str()).then_some(i)
473        }) {
474            return &self.entries[e];
475        }
476        let c = self
477            .entries
478            .iter()
479            .filter(|x| x.prefix == prefix.prefix)
480            .count();
481        let entry = LovEntry {
482            prefix: prefix.prefix.to_string(),
483            url: prefix.url.to_string(),
484            num: c,
485            state: EntryState::Transit,
486        };
487        self.entries.push(entry);
488        self.entries.last().unwrap()
489    }
490
491    pub fn save_prefix(&mut self, cache: &Cache, prefix: &Prefix, content: &str) -> Option<()> {
492        let e = self
493            .entries
494            .iter_mut()
495            .find(|e| (e.prefix == prefix.prefix && e.url == prefix.url.as_str()))?;
496        e.save(cache, content)
497    }
498}
